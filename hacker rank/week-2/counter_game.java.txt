     * The function accepts LONG_INTEGER n as parameter.
     */

    public static String counterGame(long n) {
        
        int c =0;
        long a =0;
        
        if ( n == 1){
            return "Richard";
        }
        
        while(n>1){
            a = (long)Math.pow(2, Long.toBinaryString(n).length()-1);
            n = n == a ? n /2 : n-a ;
            c++;
        }
        
        if(c % 2 == 0){
            return "Richard";
        }
        
        return "Louise";
        
    // Write your code here

    }

}
