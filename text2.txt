public class reversestring {    public static reverse(string s){        int right = s.length - 1;        int left = 0;   //this is from main branch        string temp = "";        string ans = "";        while(left<=right){            char ch = s.CharAt(left);            if (ch != ''){                temp += ch;            }            else if (ch == '') {                if(!ans.equals("")){                    ans = temp + " " + ans;                }            }            else {                ans = temp;            }            temp = "";            left++;            if(!temp.equals("")){                if(!ans.equals("")){                    ans = temp + "" + ans;                }            }            else{                ans = temp;            }        }        return ans;    }}
