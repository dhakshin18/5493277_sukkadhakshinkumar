class Result {

    /*
     * Complete the 'plusMinus' function below.
     *
     * The function accepts INTEGER_ARRAY arr as parameter.
     */

    public static void plusMinus(List<Integer> arr) {
        
         int[] intArray = arr.stream().mapToInt(Integer::intValue).toArray();
        int n= intArray.length;
        int count1=0;
        int count2=0;
        int count3=0;
        
        for(int i=0;i<n;i++){
            if (intArray[i]>0){
                count1 ++;
            }
            if (intArray[i]<0){
                count2++;
            }
            if (intArray[i] == 0){
                count3++;
            }
        }
        
        double result1 = (double)count1/n;
        double result2 = (double)count2/n;
        double result3 = (double)count3/n;
        
        System.out.printf("%.6f\n",result1);
        System.out.printf("%.6f\n",result2);
        System.out.printf("%.6f\n",result3);
    // Write your code here

    }

}

public class Solution {
    public static void main(String[] args) throws IOException {
        BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(bufferedReader.readLine().trim());

        List<Integer> arr = Stream.of(bufferedReader.readLine().replaceAll("\\s+$", "").split(" "))
            .map(Integer::parseInt)
            .collect(toList());

        Result.plusMinus(arr);

        bufferedReader.close();
    }
}
